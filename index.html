S/* ========================================== 🧠 프리미엄 모바일 청첩장 자바스크립트 (script.js) 카운트다운, 슬라이더, 아코디언, 주소/계좌 복사, 그리고 암호화 기술이 내장된 보안 방명록을 통제하는 두뇌입니다. ========================================== */ // HTML 문서가 완전히 불러와지면 작동을 시작합니다. document.addEventListener("DOMContentLoaded", () => { // ========================================== // ⏳ 1. 결혼식 디데이 카운트다운 타이머 (실시간 애니메이션 고도화) // ========================================== // 실제 결혼식 날짜인 2026년 10월 10일 오후 1시로 날짜를 타겟팅합니다! const weddingDate = new Date("2026-10-10T13:00:00").getTime(); // 숫자가 바뀔 때 은은하게 위아래로 회전하는 롤링 애니메이션 효과를 주입하는 특수 함수입니다. function setTimeValueWithAnimation(elementId, newValue) { const element = document.getElementById(elementId); if (!element) return; const oldValue = element.innerText; // 이전 숫자와 새로운 숫자가 다를 때만 움직임(애니메이션)을 줍니다! if (oldValue !== newValue) { // 1. 숫자가 회전하며 사라지도록 애니메이션 클래스 주입 element.classList.add("rolling"); // 2. 애니메이션이 절반쯤 지나 숫자가 눈에 보이지 않게 되는 0.15초(150ms) 시점에 진짜 글자를 조용히 바꿉니다! setTimeout(() => { element.innerText = newValue; }, 150); // 3. 0.3초(300ms) 애니메이션이 완전히 끝나면 다음 초에 다시 돌 수 있도록 클래스를 빼 줍니다. setTimeout(() => { element.classList.remove("rolling"); }, 300); } } function updateCountdown() { const now = new Date().getTime(); // 현재 시각 const difference = weddingDate - now; // 남은 시간 계산 if (difference <= 0) { // 결혼식 시간이 지났을 때의 처리입니다 document.getElementById("days").innerText = "00"; document.getElementById("hours").innerText = "00"; document.getElementById("minutes").innerText = "00"; document.getElementById("seconds").innerText = "00"; document.querySelector(".countdown-title").innerText = "행복하게 잘 살겠습니다! ♥"; return; } // 남은 시간 단위를 계산합니다 const days = Math.floor(difference / (1000 * 60 * 60 * 24)); const hours = Math.floor((difference % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)); const minutes = Math.floor((difference % (1000 * 60 * 60)) / (1000 * 60)); const seconds = Math.floor((difference % (1000 * 60)) / 1000); // 계산된 숫자를 애니메이션 효과와 함께 화면에 채워넣어 줍니다. // 숫자가 10보다 작을 때는 앞에 0을 붙여 자릿수를 맞춰줍니다. (예: 9초 -> 09초) const daysStr = String(days).padStart(2, "0"); const hoursStr = String(hours).padStart(2, "0"); const minutesStr = String(minutes).padStart(2, "0"); const secondsStr = String(seconds).padStart(2, "0"); // 애니메이션 효과를 동반하여 매 초 갱신! setTimeValueWithAnimation("days", daysStr); setTimeValueWithAnimation("hours", hoursStr); setTimeValueWithAnimation("minutes", minutesStr); setTimeValueWithAnimation("seconds", secondsStr); } // 처음 켜자마자 카운트다운을 한번 실행하고, 이후 1초마다 반복적으로 실행해 줍니다. updateCountdown(); setInterval(updateCountdown, 1000); // ========================================== // 📸 2. 포토 갤러리 이미지 슬라이더 // ========================================== const sliderWrapper = document.getElementById("slider-wrapper"); const slides = document.querySelectorAll(".slide"); const dots = document.querySelectorAll(".dot"); const prevBtn = document.getElementById("prev-btn"); const nextBtn = document.getElementById("next-btn"); let currentSlideIndex = 0; // 현재 보여지는 사진 번호 (0번이 첫 사진) const totalSlides = slides.length; // 총 사진 개수 function showSlide(index) { // 인덱스가 범위를 벗어나지 않도록 조절해 줍니다. if (index < 0) { currentSlideIndex = totalSlides - 1; // 첫 사진에서 이전을 누르면 마지막 사진으로 } else if (index >= totalSlides) { currentSlideIndex = 0; // 마지막 사진에서 다음을 누르면 첫 사진으로 } else { currentSlideIndex = index; } // 사진 틀을 가로로 이동시켜 사진을 바꿉니다. (예: 1번째 사진이면 100% 만큼 왼쪽으로 밉니다) sliderWrapper.style.transform = `translateX(-${currentSlideIndex * 100}%)`; // 하단 점 표시기(Indicator)의 불빛을 조절합니다. dots.forEach((dot, idx) => { if (idx === currentSlideIndex) { dot.classList.add("active"); } else { dot.classList.remove("active"); } }); } // '다음 사진' 화살표를 누르면 nextBtn.addEventListener("click", () => { showSlide(currentSlideIndex + 1); }); // '이전 사진' 화살표를 누르면 prevBtn.addEventListener("click", () => { showSlide(currentSlideIndex - 1); }); // 하단의 점들을 직접 눌러서 사진을 바꿀 때 dots.forEach((dot) => { dot.addEventListener("click", (e) => { const index = parseInt(e.target.getAttribute("data-index")); showSlide(index); }); }); // ========================================== // 📍 3. 주소 및 계좌번호 원클릭 복사 // ========================================== // 예식장 주소 복사하기 const copyAddressBtn = document.getElementById("copy-address-btn"); copyAddressBtn.addEventListener("click", () => { const addressText = document.getElementById("venue-address").innerText; navigator.clipboard.writeText(addressText).then(() => { alert("예식장 주소가 복사되었습니다. 지도 앱에 붙여넣으세요! 🗺️"); }).catch((err) => { console.error("복사 실패: ", err); }); }); // 축의금 계좌번호 복사하기 const copyAccBtns = document.querySelectorAll(".copy-acc-btn"); copyAccBtns.forEach((btn) => { btn.addEventListener("click", (e) => { // 버튼에 저장된 데이터 속성(data-account)에서 계좌번호를 가져옵니다 const accNum = e.target.getAttribute("data-account"); navigator.clipboard.writeText(accNum).then(() => { alert("계좌번호가 복사되었습니다. 편하게 송금하세요! 💳"); }).catch((err) => { console.error("복사 실패: ", err); }); }); }); // ========================================== // ✉️ 4. 마음 전하실 곳 계좌 안내 (아코디언 토글) // ========================================== const accordionHeaders = document.querySelectorAll(".accordion-header"); accordionHeaders.forEach((header) => { header.addEventListener("click", () => { const content = header.nextElementSibling; // 바로 아래에 붙어있는 계좌 내용 상자 // 이미 열려있다면 닫아줍니다. if (header.classList.contains("active")) { header.classList.remove("active"); content.classList.remove("open"); content.style.maxHeight = null; } else { // 열기 전에 다른 아코디언은 다 닫아주어 깔끔함을 유지합니다. accordionHeaders.forEach((otherHeader) => { otherHeader.classList.remove("active"); const otherContent = otherHeader.nextElementSibling; otherContent.classList.remove("open"); otherContent.style.maxHeight = null; }); // 선택한 것만 예쁘게 열어줍니다. header.classList.add("active"); content.classList.add("open"); // 아코디언이 부드럽게 늘어날 수 있도록 스크롤 높이를 높이값으로 지정해 줍니다. content.style.maxHeight = content.scrollHeight + "px"; } }); }); // ========================================== // 🔒 5. 강력한 보안이 탑재된 방명록 구현 (SHA-256 암호화) // ========================================== const guestbookForm = document.getElementById("guestbook-form"); const guestNameInput = document.getElementById("guest-name"); const guestPasswordInput = document.getElementById("guest-password"); const guestMessageInput = document.getElementById("guest-message"); const guestbookList = document.getElementById("guestbook-list"); // 로컬 저장소(localStorage)에서 기존 방명록 글 목록을 가져옵니다. // 만약 아무 글도 없다면 빈 상자([])를 준비합니다. let guestbookItems = JSON.parse(localStorage.getItem("wedding_guestbook")) || []; // [강력한 보안 암호화 함수] // 사용자가 입력한 비밀번호 원본(예: "1234")을 해커가 절대 알아볼 수 없도록 // SHA-256이라는 수학적인 금고 방식으로 암호화하여 64글자의 외계어 해시값으로 변환합니다. async function encryptSHA256(plainText) { const encoder = new TextEncoder(); const data = encoder.encode(plainText); // 비밀번호 문자열을 컴퓨터 숫자로 변환 const hashBuffer = await crypto.subtle.digest("SHA-256", data); // 표준 SHA-256 암호화 수행 const hashArray = Array.from(new Uint8Array(hashBuffer)); // 16진수 문자열로 변환하여 리턴합니다. const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join(''); return hashHex; } // 화면에 저장되어 있는 방명록 카드를 쫙 그려주는 함수입니다. function renderGuestbook() { guestbookList.innerHTML = ""; // 기존 목록을 한 번 깨끗하게 청소합니다 if (guestbookItems.length === 0) { // 아무 글도 없을 때는 아늑한 안내 문구를 띄웁니다. guestbookList.innerHTML = `

첫 번째 축하 메시지의 주인공이 되어주세요! 🌸

`; return; } // 최신 글이 맨 위로 올라오도록 시간 역순으로 카드를 그립니다. guestbookItems.slice().reverse().forEach((item) => { const card = document.createElement("div"); card.className = "guestbook-card"; // 보안을 위해 비밀번호는 이 카드에 노출되거나 저장되지 않으므로 안심하셔도 됩니다! card.innerHTML = `
${escapeHTML(item.name)} ${item.date}
${escapeHTML(item.message)}
`; guestbookList.appendChild(card); }); // 모든 삭제 버튼에 클릭 기능을 활성화합니다. setupDeleteButtons(); } // 나쁜 하객이 글 상자에 이상한 스크립트를 주입하지 못하게 하는 안전한 텍스트 변환 필터입니다. function escapeHTML(str) { return str .replace(/&/g, "&") .replace(//g, ">") .replace(/"/g, """) .replace(/'/g, "'"); } // 방명록 등록 폼을 전송(제출)했을 때 실행되는 함수입니다. guestbookForm.addEventListener("submit", async (e) => { e.preventDefault(); // 전송 후 화면이 새로고침되는 것을 방지합니다. const name = guestNameInput.value.trim(); const rawPassword = guestPasswordInput.value; // 사용자가 입력한 날것의 비밀번호 const message = guestMessageInput.value.trim(); if (!name || !rawPassword || !message) { alert("성함, 비밀번호, 내용을 모두 채워주세요!"); return; } // [최우선 보안 보안 암호화] // 비밀번호 원본은 절대로 보관하지 않습니다! 오직 암호화된 안전한 해시값만 저장합니다. const encryptedPassword = await encryptSHA256(rawPassword); // 날짜를 구합니다 (예: 2026-06-02 10:30) const now = new Date(); const dateString = `${now.getFullYear()}-${String(now.getMonth() + 1).padStart(2, '0')}-${String(now.getDate()).padStart(2, '0')} ${String(now.getHours()).padStart(2, '0')}:${String(now.getMinutes()).padStart(2, '0')}`; // 새로운 방명록 글 객체를 만듭니다. const newGuestMessage = { id: Date.now().toString(), // 고유한 식별용 번호 name: name, passwordHash: encryptedPassword, // 꽁꽁 숨겨진 비밀번호 암호화 값 저장! message: message, date: dateString }; // 기존 방명록 리스트에 새 글을 추가하고 저장합니다. guestbookItems.push(newGuestMessage); localStorage.setItem("wedding_guestbook", JSON.stringify(guestbookItems)); // 입력 칸들을 깨끗하게 비워줍니다. guestNameInput.value = ""; guestPasswordInput.value = ""; guestMessageInput.value = ""; // 새롭게 리스트를 다시 그려주고 하객에게 알립니다. renderGuestbook(); alert("축하 메시지가 등록되었습니다! 고맙습니다. 😊"); }); // 방명록 글을 삭제하는 버튼들의 기능을 관리합니다. function setupDeleteButtons() { const deleteButtons = document.querySelectorAll(".delete-card-btn"); deleteButtons.forEach((btn) => { btn.addEventListener("click", async (e) => { // 클릭한 카드의 고유 식별 번호를 가져옵니다 const idToDelete = btn.getAttribute("data-id"); // 해당 글 정보를 전체 리스트에서 찾아냅니다. const targetItem = guestbookItems.find(item => item.id === idToDelete); if (!targetItem) return; // 하객에게 비밀번호를 물어봅니다. const inputPassword = prompt("이 글을 남길 때 입력하셨던 비밀번호를 입력해 주세요:"); if (inputPassword === null) return; // 취소를 눌렀을 때는 아무 동작 안 함 // 하객이 입력한 비밀번호도 마찬가지로 같은 SHA-256 방식으로 암호화하여 대조합니다. const encryptedInputPassword = await encryptSHA256(inputPassword); // 암호화된 값끼리 비교하므로, 비밀번호 원본 노출 없이 완벽하고 안전하게 본인 검증이 완료됩니다! if (encryptedInputPassword === targetItem.passwordHash) { if (confirm("정말로 이 축하글을 삭제하시겠습니까?")) { // 비밀번호가 맞았으므로 해당 글을 리스트에서 도려냅니다. guestbookItems = guestbookItems.filter(item => item.id !== idToDelete); // 최신화된 리스트를 저장소에 갱신합니다. localStorage.setItem("wedding_guestbook", JSON.stringify(guestbookItems)); // 다시 화면을 그려줍니다. renderGuestbook(); alert("축하글이 성공적으로 삭제되었습니다."); } } else { alert("비밀번호가 일치하지 않습니다. 다시 확인해 주세요. 🔒"); } }); }); } // 최초 실행 시 저장된 방명록을 화면에 출력합니다. renderGuestbook(); });
